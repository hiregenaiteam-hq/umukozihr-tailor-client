```tsx
import { useState, useEffect } from 'react';
import ProfileForm from '../components/ProfileForm';
import JDInput from '../components/JDInput';
import JobCard from '../components/JobCard';
import LoginForm from '../components/LoginForm';
import { generation, auth } from '../lib/api';

export default function Home() {
  const [isAuthenticated, setIsAuthenticated] = useState(false);
  const [profile, setProfile] = useState(null);
  const [jobs, setJobs] = useState([]);
  const [loading, setLoading] = useState(false);
  const [currentRun, setCurrentRun] = useState(null);
  const [pollInterval, setPollInterval] = useState(null);
  const [artifacts, setArtifacts] = useState([]);

  useEffect(() => {
    // check if token exists
    const token = localStorage.getItem('token');
    if (token) {
      setIsAuthenticated(true);
    }
  }, []);

  useEffect(() => {
    // poll for status when we have a run
    if (currentRun && !pollInterval) {
      const interval = setInterval(async () => {
        try {
          const res = await generation.getStatus(currentRun);
          
          if (res.data.status === 'completed') {
            setArtifacts(res.data.artifacts);
            setLoading(false);
            setCurrentRun(null);
            clearInterval(interval);
            setPollInterval(null);
          } else if (res.data.status === 'failed') {
            alert('Generation failed: ' + res.data.error);
            setLoading(false);
            setCurrentRun(null);
            clearInterval(interval);
            setPollInterval(null);
          }
        } catch (error) {
          console.error('Poll error:', error);
        }
      }, 2000); // poll every 2 seconds
      
      setPollInterval(interval);
    }

    return () => {
      if (pollInterval) clearInterval(pollInterval);
    };
  }, [currentRun]);

  const handleLogin = (token) => {
    setIsAuthenticated(true);
  };

  const handleLogout = () => {
    localStorage.removeItem('token');
    setIsAuthenticated(false);
    setProfile(null);
    setJobs([]);
    setArtifacts([]);
  };

  const handleGenerate = async () => {
    if (!profile || jobs.length === 0) {
      alert('Please add your profile and at least one job');
      return;
    }

    setLoading(true);
    setArtifacts([]);

    try {
      const res = await generation.generate(profile, jobs);
      setCurrentRun(res.data.run_id);
    } catch (error) {
      console.error('Generation error:', error);
      alert('Failed to start generation');
      setLoading(false);
    }
  };

  if (!isAuthenticated) {
    return (
      <div className="min-h-screen bg-gray-50">
        <div className="max-w-7xl mx-auto p-8">
          <h1 className="text-4xl font-bold text-center mb-8">UmukoziHR Resume Tailor</h1>
          <LoginForm onLogin={handleLogin} />
        </div>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-gray-50">
      <div className="max-w-7xl mx-auto p-8">
        <div className="flex justify-between items-center mb-8">
          <h1 className="text-4xl font-bold">UmukoziHR Resume Tailor</h1>
          <button
            onClick={handleLogout}
            className="px-4 py-2 bg-gray-600 text-white rounded hover:bg-gray-700"
          >
            Logout
          </button>
        </div>

        <div className="grid grid-cols-1 lg:grid-cols-2 gap-8">
          <div>
            <h2 className="text-2xl font-semibold mb-4">Your Profile</h2>
            <ProfileForm onProfileChange={setProfile} />
          </div>

          <div>
            <h2 className="text-2xl font-semibold mb-4">Job Descriptions</h2>
            <JDInput onJobsChange={setJobs} />
          </div>
        </div>

        <div className="mt-8 text-center">
          <button
            onClick={handleGenerate}
            disabled={loading || !profile || jobs.length === 0}
            className={`px-8 py-4 text-white rounded-lg text-lg font-semibold ${
              loading || !profile || jobs.length === 0
                ? 'bg-gray-400 cursor-not-allowed'
                : 'bg-blue-600 hover:bg-blue-700'
            }`}
          >
            {loading ? (
              <span className="flex items-center">
                <svg className="animate-spin h-5 w-5 mr-3" viewBox="0 0 24 24">
                  <circle className="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" strokeWidth="4" fill="none" />
                  <path className="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z" />
                </svg>
                Processing... (Run ID: {currentRun?.slice(0, 8)}...)
              </span>
            ) : (
              'Generate Tailored Documents'
            )}
          </button>
        </div>

        {artifacts && Object.keys(artifacts).length > 0 && (
          <div className="mt-8">
            <h2 className="text-2xl font-semibold mb-4">Generated Documents</h2>
            <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
              {Object.entries(artifacts).map(([key, url]) => (
                <JobCard key={key} artifact={{ name: key, url }} />
              ))}
            </div>
          </div>
        )}
      </div>
    </div>
  );
}
```

```tsx
import { useState, useEffect } from 'react';

export default function ProfileForm({ onProfileChange }) {
  const [profile, setProfile] = useState({
    name: '',
    contacts: {
      email: '',
      phone: '',
      location: '',
      links: ['']
    },
    summary: '',
    skills: [''],
    experience: [],
    education: [],
    projects: []
  });

  useEffect(() => {
    // load saved profile from localStorage
    const savedProfile = localStorage.getItem('userProfile');
    if (savedProfile) {
      const parsed = JSON.parse(savedProfile);
      setProfile(parsed);
      onProfileChange(parsed);
    }
  }, []);

  const handleChange = (field, value) => {
    const updated = { ...profile, [field]: value };
    setProfile(updated);
    onProfileChange(updated);
    // save to localStorage
    localStorage.setItem('userProfile', JSON.stringify(updated));
  };

  const handleContactChange = (field, value) => {
    const updated = {
      ...profile,
      contacts: { ...profile.contacts, [field]: value }
    };
    setProfile(updated);
    onProfileChange(updated);
    localStorage.setItem('userProfile', JSON.stringify(updated));
  };

  return (
    <div className="space-y-4 bg-white p-6 rounded-lg shadow">
      <input
        type="text"
        placeholder="Full Name"
        value={profile.name}
        onChange={(e) => handleChange('name', e.target.value)}
```

```tsx
import { useState } from 'react';

export default function JDInput({ onJobsChange }) {
  const [jobs, setJobs] = useState([]);
  const [currentJob, setCurrentJob] = useState({
    company: '',
    title: '',
    region: 'US',
    jd_text: ''
  });

  const handleAdd = () => {
    if (!currentJob.company || !currentJob.title || !currentJob.jd_text) {
      alert('Please fill all fields');
      return;
    }

    const newJob = {
      ...currentJob,
      id: `${currentJob.company}-${currentJob.title}-${Date.now()}`
    };

    const updatedJobs = [...jobs, newJob];
    setJobs(updatedJobs);
    onJobsChange(updatedJobs);
    
    // reset form
    setCurrentJob({
      company: '',
      title: '',
      region: 'US',
      jd_text: ''
    });
  };

  const handleRemove = (id) => {
    const updated = jobs.filter(j => j.id !== id);
    setJobs(updated);
    onJobsChange(updated);
  };

  return (
    <div className="space-y-4 bg-white p-6 rounded-lg shadow">
      <input
        type="text"
        placeholder="Company Name"
        value={currentJob.company}
        onChange={(e) => setCurrentJob({ ...currentJob, company: e.target.value })}
        className="w-full p-2 border rounded"
      />
      
      <input
        type="text"
        placeholder="Job Title"
        value={currentJob.title}
        onChange={(e) => setCurrentJob({ ...currentJob, title: e.target.value })}
        className="w-full p-2 border rounded"
      />
      
      <select
        value={currentJob.region}
        onChange={(e) => setCurrentJob({ ...currentJob, region: e.target.value })}
        className="w-full p-2 border rounded"
      >
        <option value="US">US Format</option>
        <option value="EU">EU Format</option>
        <option value="GL">Global Format</option>
      </select>
      
      <textarea
        placeholder="Paste Job Description"
        value={currentJob.jd_text}
        onChange={(e) => setCurrentJob({ ...currentJob, jd_text: e.target.value })}
        className="w-full p-2 border rounded h-32"
      />
      
      <button
        onClick={handleAdd}
        className="w-full px-4 py-2 bg-green-600 text-white rounded hover:bg-green-700"
      >
        Add Job
      </button>

      {jobs.length > 0 && (
        <div className="mt-4 space-y-2">
          <h3 className="font-semibold">Added Jobs ({jobs.length})</h3>
          {jobs.map(job => (
            <div key={job.id} className="flex justify-between items-center p-2 bg-gray-50 rounded">
              <span>{job.company} - {job.title}</span>
              <button
                onClick={() => handleRemove(job.id)}
                className="text-red-600 hover:text-red-800"
              >
                Remove
              </button>
            </div>
          ))}
        </div>
      )}
    </div>
  );
}
```
